65 def process_univariate_data(self, win: int) -> None:
66 """
67 Processes the univariate data.
68
69 Parameters:
70 -----------
71 win : int
72 Window size.
73 """
74 for ahead in self._lahead:
75 df = self.df
76
77 win_x = sliding_window_view(df['PX_LAST'].to_numpy(), window_shape=win)[::1]
78 Y = df.iloc[ahead + win:]['PX_LAST']
79 X = pd.DataFrame.from_records(win_x)
80
81 X = X.iloc[:-(ahead + 1), :]
82 X.set_index(df.index[(win - 1):-(ahead + 1)], drop=True, inplace=True)
83
84 x_mean = self.calculate_mean(X, axis=1)
85 center_x = X.sub(x_mean, axis=0)
86
87 min_x = min(self.calculate_min(center_x))
88 max_x = max(self.calculate_max(center_x))
89 vdd = pd.DataFrame({'mean': x_mean, 'min': min_x, 'max': max_x})
90 vdd.set_index(X.index)
91
92 center_x_norm = center_x.apply(lambda x: (x - min_x) / (max_x - min_x), axis=1)
93 center_y_norm = pd.Series([((i - j) - min_x) / (max_x - min_x)
94 for i, j in zip(Y.tolist(), x_mean.tolist())], index=Y.index)
95 center_x_norm = center_x_norm.astype('float32')
96 center_y_norm = center_y_norm.astype('float32')
97
98 pmod = int(center_x_norm.shape[0] * self.tr_tst)
99 train_x = center_x_norm.iloc[:pmod, :]
100 train_y = center_y_norm.iloc[:pmod]
101 test_x = center_x_norm.iloc[pmod:, :]
102 test_y = center_y_norm.iloc[pmod:]
103 self.serial_dict['INPUT_DATA'][ahead] = {
104 "x": X, "y": Y, "nx": center_x_norm,
105 "ny": center_y_norm, "numt": pmod,
106 "trainX": train_x, "trainY": train_y,
107 "testX": test_x, "testY": test_y, "vdd": vdd
108 }

110 def process_multivariate_data(self, mwin: int, m_ftrs: int) -> None:
111 """
112 Processes the multivariate data.
113
114 Parameters:
115 -----------
116 mwin : int
117 Multivariate window size.
118 m_ftrs : int
119 Multivariate number of features.
120 """
121 self.check_infinite_values(self.df)
122
123 df_y = self.df["PX_LAST"].copy()
124 df_x = self.df[["PX_LAST", "PX_OPEN", "RSI_14D", "PX_TREND",
125 "PX_VTREND", "TWEET_POSTIVIE", "TWEET_NEGATIVE",
126 "NEWS_POSITIVE", "NEWS_NEGATIVE", "VOLATILITY"]].copy()
127 idx = df_x.index[(mwin - 1):]
128
129 multi_x, cols = self.prepare_multivariate_data(df_x, mwin, m_ftrs, idx)
130
131 for ahead in self._lahead:
132 multi_x_list = multi_x[:-(ahead + 1), :, :]
133 multi_y_list = df_y.iloc[ahead + mwin:]
134 mean_multi_x = self.calculate_mean(multi_x_list, axis=1)
135
136 mean_multi_x_c = multi_x_list - mean_multi_x[:, None, :]
137 pavX = pd.DataFrame(mean_multi_x, columns=df_x.columns)
138 pavX.set_index(df_x.index[(mwin - 1):-(ahead + 1)], drop=True, inplace=True)
139 max_mult_x = self.calculate_max(mean_multi_x_c, axis=(0, 1))
140 min_mult_x = self.calculate_min(mean_multi_x_c, axis=(0, 1))
141 mvdd = {"mean": pavX, "min": min_mult_x, "max": max_mult_x}
142 multi_x_norm = (mean_multi_x_c - min_mult_x[None, None, :]) / (max_mult_x[None, None, :] - min_mult_x[None, None, :] + 0.00001)
143 multi_y_norm = ((multi_y_list.to_numpy() - mean_multi_x[:, 0]) - min_mult_x[0]) / (max_mult_x[0] - min_mult_x[0] + 0.00001)
144
145 multi_x_norm = multi_x_norm.astype("float32")
146 multi_y_norm = multi_y_norm.astype("float32")
147
148 multi_train_x, multi_train_y, multi_test_x, multi_test_y, pmod = self.prepare_data_for_modeling(multi_x_norm, multi_y_norm, multi=True)
149
150 self.check_nan_values(multi_train_x, multi_test_x, ahead)
151
152 xdx = idx[:-(ahead + 1)]
153 self.mserial_dict['INPUT_DATA'][ahead] = {"x": multi_x_list, "y": multi_y_list, "nx": multi_x_norm, "ny": multi_y_norm, "numt": pmod,
154 "trainX": multi_train_x, "trainY": multi_train_y,
155 "testX": multi_test_x, "testY": multi_test_y, "vdd": mvdd, "cnms": cols,
156 "idtest": xdx[pmod:]}

158 def prepare_multivariate_data(self, df_x: pd.DataFrame, win: int, n_ftrs: int, idx: pd.Index) -> tuple:
159 """
160 Prepares the multivariate data for modeling.
161
162 Parameters:
163 -----------
164 df_x : pd.DataFrame
165 The features.
166 win : int
167 The window size.
168 n_ftrs : int
169 The number of features.
170 idx : pd.Index
171 The index.
172
173 Returns:
174 --------
175 Tuple[np.ndarray, pd.Index]
176 The multivariate data and columns.
177 """
178 multi_x = []
179 cols = df_x.columns
180 for i in range(len(df_x.columns)):
181 lX = np.lib.stride_tricks.sliding_window_view(df_x.iloc[:, i].to_numpy(), window_shape=win)[::n_ftrs]
182 ss = pd.DataFrame.from_records(lX)
183 ss.set_index(idx, drop=True, inplace=True)
184 multi_x.append(ss.to_numpy())
185 multi_x = np.transpose(np.stack(multi_x, axis=1), (0, 2, 1))
186 return multi_x, cols

188 def compute_sentiment_scores(self) -> None:
189 """
190 Computes the sentiment scores for the data.
191 """
192 self.df['TWEET_POSTIVIE'] = self.df['TWITTER_POS_SENTIMENT_COUNT'] / self.df['TWITTER_PUBLICATION_COUNT'].replace(0, 1)
193 self.df['TWEET_NEGATIVE'] = self.df['TWITTER_NEG_SENTIMENT_COUNT'] / self.df['TWITTER_PUBLICATION_COUNT'].replace(0, 1)
194 self.df['NEWS_POSITIVE'] = self.df['NEWS_POS_SENTIMENT_COUNT'] / self.df['NEWS_PUBLICATION_COUNT'].replace(0, 1)
195 self.df['NEWS_NEGATIVE'] = self.df['NEWS_NEG_SENTIMENT_COUNT'] / self.df['NEWS_PUBLICATION_COUNT'].replace(0, 1)

203 def compute_trend(self) -> None:
204 '''
205 Computes the trend for the price and volume data
206 '''
207 self.df['PX_TREND'] = 2 * (self.df['PX_LAST'] - self.df['PX_OPEN']) / (self.df['PX_OPEN'] + self.df['PX_LAST'])
208 self.df['PX_VTREND'] = self.df['PX_TREND'] * self.df['VOLUME']

244 def prepare_data_for_modeling(self, x_norm, y_norm, multi: bool) -> tuple:
245 '''
246 Returns the data for modeling
247
248 Parameters:
249 x_norm : the features
250 y_norm : the target
251 multi : if the data is multivariate
252
253 Returns:
254 Train_x : data for training
255 Train_y : target for training
256 Test_x : data for testing
257 Test_y : target for testing
258 '''
259 pmod = int(x_norm.shape[0] * self.tr_tst)
260 if multi is False: # Univariate
261 train_x = x_norm.iloc[:pmod,:]
262 train_y = y_norm.iloc[:pmod]
263 test_x = x_norm.iloc[pmod:,:]
264 test_y = y_norm.iloc[pmod:]
265 else: # Multivariate
266 train_x = x_norm[:pmod, :, :]
267 train_y = y_norm[:pmod]
268 test_x = x_norm[pmod:, :, :]
269 test_y = y_norm[pmod:]
270 return train_x, train_y, test_x, test_y, pmod
