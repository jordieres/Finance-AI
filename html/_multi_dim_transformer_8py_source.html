25class TransformerL(nn.Module):
26 """
27 Linear transformer-based model for time series forecasting.
28 """
29 def __init__(self, input_dim: int, embed_dim: int, num_layers: int, num_heads: int, dropout: float):
30 super(TransformerL, self).__init__()
31 self.input_dim = input_dim
32 self.embed_dim = embed_dim
33 self.num_layers = num_layers
34 self.num_heads = num_heads
35 self.dropout = dropout
36
37 self.embedding = nn.Linear(self.input_dim, self.embed_dim)
38 self.encoder_layers = nn.ModuleList([nn.TransformerEncoderLayer(self.embed_dim, self.num_heads, dim_feedforward=4*self.embed_dim, dropout=self.dropout)
39 for _ in range(self.num_layers)])
40 self.output_layer = nn.Linear(self.embed_dim, 1)
41
42 def forward(self, src) -> torch.Tensor:
43 """
44 Forward pass of the Transformer model.
45
46 Parameters
47 ----------
48 src : torch.Tensor
49 Input tensor of shape (batch_size, seq_length, input_dim).
50
51 Returns
52 -------
53 torch.Tensor
54 Output tensor of shape (batch_size, seq_length, 1).
55 """
56 src_embedded = self.embedding(src)
57 src_embedded = src_embedded.permute(1, 0, 2)
58
59 for encoder in self.encoder_layers:
60 src_embedded = encoder(src_embedded)
61
62 output = self.output_layer(src_embedded)
63 return output

65def transformer_fun(transformer_parameters: dict, train_X: pd.DataFrame,
66 train_y: pd.DataFrame, test_X: pd.DataFrame, test_y: pd.DataFrame,
67 Y: pd.DataFrame, vdd: dict, epochs: int, bsize: int,
68 nhn: int, win: int, n_ftrs: int, ahead: int, stock: str,
69 test_X_idx: pd.Index, k: int, seed: int) -> dict:
70 """
71 Trains a Transformer model for time series forecasting.
72
73 Parameters
74 ----------
75 transformer_parameters : dict
76 Dictionary containing the number of layers and heads for the Transformer model.
77 train_X : pd.DataFrame
78 Training input data.
79 train_y : pd.DataFrame
80 Training target data.
81 test_X : pd.DataFrame
82 Test input data.
83 test_y : pd.DataFrame
84 Test target data.
85 Y : pd.DataFrame
86 Target data.
87 vdd : dict
88 Dictionary containing the mean, min, and max values used for normalization.
89 epochs : int
90 Number of epochs.
91 bsize : int
92 Batch size.
93 nhn : int
94 Number of hidden neurons (embedding dimension).
95 win : int
96 Window size.
97 n_ftrs : int
98 Number of features.
99 ahead : int
100 Number of days ahead.
101 stock : str
102 Stock ticker.
103 test_X_idx : pd.Index
104 Index of the test input data.
105 k : int
106 Number of features in the input data.
107 seed : int
108 Random seed.
109
110 Returns
111 -------
112 dict
113 Dictionary containing the results of the Transformer model training.
114 """
115 device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
116 # Convert to PyTorch tensors
117 np_train_X = torch.tensor(train_X.reshape(train_X.shape[1], train_X.shape[0], k),
118 dtype=torch.float32).to(device)
119 np_train_y = torch.tensor(train_y, dtype=torch.float32).to(device)
120 np_test_X = torch.tensor(test_X.reshape(test_X.shape[1], test_X.shape[0], k),
121 dtype=torch.float32).to(device)
122 np_test_y = torch.tensor(test_y, dtype=torch.float32).to(device)
123
124 nit = 0
125 lloss = float('nan')
126 model = TransformerL(input_dim=k, embed_dim=nhn,
127 num_layers=transformer_parameters['num_layers'], num_heads=transformer_parameters['num_heads'], dropout=0.01)
128 # Define the loss function and optimizer
129 model.to(device)
130 while torch.isnan(torch.Tensor([lloss])) and nit < 5:
131 torch.manual_seed(seed)
132 criterion = nn.MSELoss()
133 optimizer = optim.AdamW(model.parameters(), lr=0.0001)
134 outputs = []
135 # Train the model
136 print("Training the model...")
137 for epoch in range(epochs):
138 optimizer.zero_grad()
139 for i in range(0, len(np_train_X), bsize):
140 batch_np_train_X = np_train_X[:, i:i+bsize, :].to(device)
141 batch_np_train_y = np_train_y[i:i+bsize].to(device)
142 output = model(batch_np_train_X)
143 loss = criterion(output, batch_np_train_y)
144 loss.backward()
145 optimizer.step()
146 if epoch % 10 == 0:
147 print(f"Epoch [{epoch+1}/{epochs}]:")
148 print(f"Train Loss: {loss.item()}")
149 #Evaluate the model
150 with torch.no_grad():
151 y_hat = model(np_test_X)
152 test_loss = criterion(y_hat, np_test_y)
153 if epoch % 10 == 0:
154 print(f"Test Loss: {test_loss.item()}")
155 print('-----------------------------------')
156 outputs.append(y_hat.cpu().detach().numpy())
157 lloss = loss.item()
158 nit += 1
159
160 y_forecast = denormalize_data(outputs[0], vdd, test_X_idx, True)
161 #y_forecast = y_forecast.apply(lambda x: x) # get the value
162 y_real = Y.loc[test_X_idx] # y real
163 y_yesterday = Y.shift(ahead).loc[test_X_idx] # Y yesterday
164 DY = pd.concat([y_real,y_forecast,y_yesterday],axis=1)
165 DY.columns = ['Y_real','Y_predicted','Y_yesterday']
166
167 msep = mean_squared_error(DY.Y_predicted, DY.Y_real) # error y predicted - y real
168 msey = mean_squared_error(DY.Y_yesterday, DY.Y_real) # error y yesterday - y real
169 maep = mean_absolute_error(DY.Y_predicted , DY.Y_real) # error y predicted - y real
170 maey = mean_absolute_error(DY.Y_yesterday , DY.Y_real) # error y yesterday - y real
171 # Prepare the result dictionary
172 df_result = {
173 'MSEP': msep,'MSEY': msey,
174 'MAEP': maep, 'MAEY': maey,
175 'Stock': stock,'DY': DY,
176 'ALG': 'Transformer','seed': seed,
177 'epochs': epoch,'nhn': nhn,'win': win,
178 'ndims': 1,'lossh': lloss,
179 'nit': nit,'model': model
180 }
181
182 return df_result
