54def eval_lstm(np_test_X: np.array, np_test_y: np.array, test_X: pd.DataFrame, model, vdd: dict, Y: pd.Series, ahead: int) -> dict:
55 """Evaluates the LSTM model"""
56 y_hat = model.predict(np_test_X,verbose=0)
57 tans = y_hat.shape
58 if len(tans) > 2 and tans[1] == 1:
59 y_hat= np.concatenate(y_hat,axis=0)
60 if len(tans) > 2 and tans[1] > 1:
61 y_hat= [vk[0].tolist() for vk in y_hat]
62 y_pred = np.concatenate(y_hat,axis=0).tolist()
63 eff = pd.DataFrame({'Yorg':np_test_y,'Yfct':y_pred})
64 eff.set_index(test_X.index,drop=True,inplace=True)
65 absolute_difference = np.abs(y_pred - np_test_y)
66 correct = np.count_nonzero(absolute_difference <= 0.1)
67 accuracy = round((correct / len(np_test_y)) *100, 3)
68 print(f"Accuracy: {accuracy}%")
69 jdx = test_X.index
70 y_forecast = denormalize_data(y_pred, vdd, jdx, multi=False, lstm=True) # denormalize the forecast
71 y_real = Y.loc[jdx] # y real
72 y_yesterday = Y.shift(ahead).loc[jdx] # Y yesterday
73 DY = pd.concat([y_real,y_forecast,y_yesterday],axis=1)
74 DY.columns = ['Y_real','Y_predicted','Y_yesterday']
75
76 msep= mean_squared_error(DY.Y_predicted , DY.Y_real) # error y predicted - y real
77 msey= mean_squared_error(DY.Y_yesterday , DY.Y_real) # error y yesterday - y real
78 maep = mean_absolute_error(DY.Y_predicted , DY.Y_real) # error y predicted - y real
79 maey = mean_absolute_error(DY.Y_yesterday , DY.Y_real) # error y yesterday - y real
80
81 return({'msep':msep,'msey':msey, 'maep':maep, 'maey':maey, 'Ys':DY, 'eff': eff})

83def load_output_preprocessed_data(path: str, win: int, tr_tst: float, scenario_name: str) -> dict:
84 """Loads all pickle files from the specified directory that match the scenario name"""
85 all_results = {}
86 directory = f'{path}/output/{win}/{tr_tst}/'
87
88 # Verify if the directory exists
89 if not os.path.exists(directory):
90 raise FileNotFoundError(f"The directory {directory} does not exist.")
91
92 for file in os.listdir(directory): # Loop through each file in the directory
93 # Verify if the file starts with scenario_name and ends with 'output.pkl' or 'm-output.pkl'
94 if file.startswith(scenario_name) and file.endswith('-output.pkl'):
95 # Extract the model type correctly
96 mdl = file.split('-')[1]
97 if file.endswith('-m-output.pkl'):
98 mdl = file.split('-')[1] + '-m'
99 else:
100 continue
101
102 # Construct the full file path
103 fdat = os.path.join(directory, file)
104
105 with open(fdat, "rb") as openfile:
106 results = {}
107 while True:
108 try:
109 directory1 = pickle.load(openfile)
110 fdat1 = pickle.load(openfile)
111 lahead = pickle.load(openfile)
112 lpar = pickle.load(openfile)
113 tot_res = pickle.load(openfile)
114 results['path'] = directory1
115 results['fdat'] = fdat1
116 results['lahead'] = lahead
117 results['lpar'] = lpar
118 results['tot_res'] = tot_res
119 except EOFError:
120 break
121 all_results[mdl] = results
122
123 return all_results
