24class TransformerL(nn.Module):
25 """
26 Transformer-based model for time series forecasting.
27 """
28 def __init__(self, input_dim: int, embed_dim: int, num_layers: int, num_heads: int, dropout: float):
29 super(TransformerL, self).__init__()
30 self.input_dim = input_dim
31 self.embed_dim = embed_dim
32 self.num_layers = num_layers
33 self.num_heads = num_heads
34 self.dropout = dropout
35
36 self.embedding = nn.Linear(self.input_dim, self.embed_dim)
37 self.encoder_layers = nn.ModuleList([nn.TransformerEncoderLayer(self.embed_dim, self.num_heads, dim_feedforward=4*self.embed_dim, dropout=self.dropout)
38 for _ in range(self.num_layers)])
39 self.output_layer = nn.Linear(self.embed_dim, 1)
40
41 def forward(self, src) -> torch.Tensor:
42 """
43 Forward pass of the Transformer model.
44
45 Parameters
46 ----------
47 src : torch.Tensor
48 Input tensor of shape (batch_size, seq_length, input_dim).
49
50 Returns
51 -------
52 torch.Tensor
53 Output tensor of shape (batch_size, seq_length, 1).
54 """
55 src_embedded = self.embedding(src)
56 src_embedded = src_embedded.permute(1, 0, 2)
57
58 for encoder in self.encoder_layers:
59 src_embedded = encoder(src_embedded)
60
61 #src_embedded = src_embedded.permute(1, 0, 2) # Volvemos a la forma original (batch, seq, embed)
62 output = self.output_layer(src_embedded)
63 return output

66def transformer_fun(transformer_parameters: dict, train_X: pd.DataFrame, train_y: pd.DataFrame, test_X: pd.DataFrame, test_y: pd.DataFrame,
67 Y: pd.DataFrame, vdd: dict, epochs: int, bsize: int, nhn: int, win: int, n_ftrs: int, ahead: int, stock: str, seed: int) -> dict:
68 """
69 Trains a Transformer model for time series forecasting.
70
71 Parameters
72 ----------
73 transformer_parameters : dict
74 Dictionary containing the number of layers and heads for the Transformer model.
75 train_X : pd.DataFrame
76 Training input data.
77 train_y : pd.DataFrame
78 Training target data.
79 test_X : pd.DataFrame
80 Test input data.
81 test_y : pd.DataFrame
82 Test target data.
83 Y : pd.DataFrame
84 Target data.
85 vdd : dict
86 Dictionary containing the mean, min, and max values used for normalization.
87 epochs : int
88 Number of epochs.
89 bsize : int
90 Batch size.
91 nhn : int
92 Number of hidden neurons (embedding dimension).
93 win : int
94 Window size.
95 n_ftrs : int
96 Number of features.
97 ahead : int
98 Number of days ahead.
99 stock : str
100 Stock ticker.
101 seed : int
102 Random seed.
103
104 Returns
105 -------
106 dict
107 Dictionary containing the results of the Transformer model training.
108 """
109 # Convert to PyTorch tensors
110 device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
111 np_train_X = torch.tensor(train_X.to_numpy().reshape(train_X.shape[1], train_X.shape[0], n_ftrs),
112 dtype=torch.float32).to(device)
113 np_train_y = torch.tensor(train_y.to_numpy(), dtype=torch.float32).to(device)
114 np_test_X = torch.tensor(test_X.to_numpy().reshape(test_X.shape[1], test_X.shape[0], n_ftrs),
115 dtype=torch.float32).to(device)
116 np_test_y = torch.tensor(test_y.to_numpy(), dtype=torch.float32).to(device)
117
118 nit = 0
119 lloss = float('nan')
120 model = TransformerL(input_dim=(np_train_X.shape[2]), embed_dim=nhn,
121 num_layers=transformer_parameters['num_layers'], num_heads=transformer_parameters['num_heads'], dropout=0.05)
122 model.to(device)
123 while torch.isnan(torch.Tensor([lloss])) and nit < 5:
124 torch.manual_seed(seed)
125 # Define the loss function and optimizer
126 criterion = nn.MSELoss()
127 optimizer = optim.AdamW(model.parameters(), lr=0.001)
128
129 outputs = []
130 # Train the model
131 print("Training the model...")
132 for epoch in range(epochs):
133 optimizer.zero_grad()
134 for i in range(0, len(np_train_X), bsize):
135 batch_np_train_X = np_train_X[:, i:i+bsize, :].to(device)
136 batch_np_train_y = np_train_y[i:i+bsize].to(device)
137 output = model(batch_np_train_X)
138 loss = criterion(output, batch_np_train_y) # Unsqueeze to match output shape
139 loss.backward()
140 optimizer.step()
141 if epoch % 10 == 0:
142 print(f"Epoch [{epoch+1}/{epochs}]:")
143 print(f"Train Loss: {loss.item()}")
144 #Evaluate the model
145 with torch.no_grad():
146 y_hat = model(np_test_X)
147 test_loss = criterion(y_hat, np_test_y)
148 if epoch % 10 == 0:
149 print(f"Test Loss: {test_loss.item()}")
150 print('-----------------------------------')
151 outputs.append(y_hat.cpu().detach().numpy())
152 lloss = loss.item()
153 nit += 1
154
155 jdx = test_X.index
156 y_forecast = denormalize_data(outputs[0], vdd, jdx, False) # denormalize the forecast
157 y_real = Y.loc[jdx] # y real
158 y_yesterday = Y.shift(ahead).loc[jdx] # Y yesterday
159 DY = pd.concat([y_real,y_forecast,y_yesterday],axis=1)
160 DY.columns = ['Y_real','Y_predicted','Y_yesterday']
161 msep = mean_squared_error(DY.Y_predicted , DY.Y_real) # error y predicted - y real
162 msey = mean_squared_error(DY.Y_yesterday , DY.Y_real) # error y yesterday - y real
163 maep = mean_absolute_error(DY.Y_predicted , DY.Y_real) # error y predicted - y real
164 maey = mean_absolute_error(DY.Y_yesterday , DY.Y_real) # error y yesterday - y real
165 # Prepare the result dictionary
166 df_result = {
167 'MSEP': msep,'MSEY': msey,
168 'MAEP': maep, 'MAEY': maey,
169 'Stock': stock,'DY': DY,
170 'ALG': 'Transformer','seed': seed,
171 'epochs': epoch,'nhn': nhn,'win': win,
172 'ndims': 1,'lossh': lloss,
173 'nit': nit,'model': model
174 }
175
176 return df_result
