48 def lstm_fun(self, ahead: int, seed: int) -> dict:
49 """Trains a basic LSTM model and evaluates its performance.
50
51 Args:
52 ahead (int): Number of days ahead for forecasting.
53 seed (int): Random seed for reproducibility.
54
55 Returns:
56 dict: Results of the LSTM model training and evaluation.
57 """
58 nptrX = self.trainX.to_numpy().reshape(
59 self.trainX.shape[0],self.trainX.shape[1],self.n_ftrs)
60 nptrY = self.trainY.to_numpy()
61 nit = 0
62 lloss= np.nan
63 while math.isnan(lloss) and nit < 5:
64 tf.random.set_seed(seed)
65 model = Sequential()
66 model.add(LSTM(self.nhn, activation='relu', input_shape=(self.win, self.n_ftrs)))
67 model.add(Dense(self.n_ftrs)) # Output of a single value
68 model.compile(loss='mean_squared_error', optimizer='adam')
69 checkpoint = ModelCheckpoint('model-basic.keras', verbose=1, \
70 monitor='val_loss',save_best_only=True, mode='auto')
71 hist = model.fit(nptrX, nptrY, epochs=self.epoch, batch_size=self.bsize, verbose=0, callbacks=[checkpoint])
72 lloss= hist.history['loss'][-1]
73 nit = nit + 1
74 # Predict
75 nptstX = self.testX.to_numpy().reshape(
76 self.testX.shape[0],self.testX.shape[1],self.n_ftrs)
77 nptstY = self.testY.to_numpy()
78 res1 = eval_lstm(nptstX, nptstY, self.testX,
79 model, self.vdd, self.Y, ahead)
80 df_result = {'MSEP':res1.get("msep"),'MSEY': res1.get("msey"),
81 'MAEP':res1.get("maep"),'MAEY': res1.get("maey"),
82 'Stock':self.stock, 'DY':res1.get("Ys"),'nhn':self.nhn,
83 'win':self.win,'ndims':1, 'lossh':lloss, 'nit':nit,
84 'model':model}
85 return(df_result)

130 def att_lstm_fun(self, ahead: int, seed: int) -> dict:
131 """Trains an attention-based LSTM model and evaluates its performance.
132
133 Args:
134 ahead (int): Number of days ahead for forecasting.
135 seed (int): Random seed for reproducibility.
136
137 Returns:
138 dict: Results of the attention-based LSTM model training and evaluation.
139 """
140 aptrX = self.trainX.to_numpy().reshape(
141 self.trainX.shape[0], self.win,self.n_ftrs)
142 aptrY = self.trainY.to_numpy()
143 nit = 0
144 lloss= np.nan
145 while math.isnan(lloss) and nit < 5:
146 amodel = Sequential()
147 amodel.add(Embedding(input_dim=self.win,output_dim=int(1),
148 mask_zero=True))
149 amodel.add(GRU(self.nhn, activation='relu', \
150 return_sequences=True,input_shape=(self.win,self.n_ftrs)))
151 amodel.add(SeqSelfAttention(attention_activation='sigmoid'))
152 amodel.add(Dense(self.n_ftrs))
153 amodel.compile(loss='mean_squared_error', optimizer='adam')
154 earlyStopping = EarlyStopping(monitor='val_loss', \
155 patience=10, verbose=0, mode='min')
156 checkpoint = ModelCheckpoint('model-attention.h5', verbose=1, \
157 monitor='val_loss',save_best_only=True, mode='auto')
158 hist = amodel.fit(aptrX, aptrY, epochs=self.epoch, callbacks=[checkpoint], \
159 validation_split=0.15,batch_size=self.bsize,verbose=1)
160 lloss= hist.history['val_loss'][-1]
161 nit = nit + 1
162 # Predict
163 amodel = load_model('model-attention.h5',custom_objects={
164 'SeqSelfAttention': SeqSelfAttention})
165 aptstX = self.testX.to_numpy().reshape(
166 self.testX.shape[0],self.win,self.n_ftrs)
167 aptstY = self.testY.to_numpy()
168 res5b = eval_lstm(aptstX, aptstY, self.testX, amodel, self.vdd, self.Y, ahead)
169 #
170 df_result = {'MSEP': res5b.get("msep"),'MSEY': res5b.get("msey"),
171 'MAEP':res5b.get("maep"),'MAEY': res5b.get("maey"),
172 'Stock': self.stock,'DY':res5b.get("Ys"),
173 'ALG':'ATT_LSTM',
174 'seed':seed,'epochs':self.epoch,'nhn':self.nhn,
175 'win':self.win, 'ndims':1,'lossh':lloss, 'nit':nit,
176 'model':amodel}
